<script type="text/javascript">
function loadstream(stream) {
    console.log(stream);
    video = document.getElementById("webcam");
    video.src = window.URL.createObjectURL(stream);
    //video.play(); works, but moved to video ready handler
    video.onloadedmetadata = video_loaded;
}

function error(error) {
    console.log(error);
}

function video_loaded(e) {
    console.log("video loaded");
    console.log(e);
    video = document.getElementById("webcam");
    video.play();
}

function start() {
    /* API reference: 
    ## Webcam html5

constraints = {video: true} if(navigator.MediaDevices) navigator.MediaDevices.getUserMedia(constraints) .then(loadstream) .catch(error); else { navigator.getMedia = ( navigator.getUserMedia || navigator.webkitGetUserMedia || navigator.mozGetUserMedia || navigator.msGetUserMedia);

        navigator.getMedia(constraints, loadstream, error);
    }
}
